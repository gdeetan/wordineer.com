**Goal:** Build a full-page Danish name generator at `/random-danish-name-generator/` with era-aware patronymic surnames, 5 filters, and a dedicated dataset of 150+ given names and 100+ surnames.

**Architecture:** Follows the Scottish name generator pattern exactly (`template-deploy/tools-src/random-scottish-name-generator.html` + `wordineer-deploy/data/scottish-names.json`). **Tech Stack:** Vanilla JS, static HTML/CSS, JSON data file, `python3 build.py` build step, Cloudflare Pages deploy. Test runner: `node tests/[file].test.js`.

## Data Schema

### `danish-names.json`

```json
{
  "given": [
    {
      "name": "Bjørn",
      "gender": "m",
      "eras": ["viking", "medieval"],
      "meaning": "bear; strength and courage"
    }
  ],
  "surnames": [
    {
      "type": "fixed",
      "name": "Jensen",
      "eras": ["classic", "modern"],
      "meaning": "son of Jens"
    },
    {
      "type": "patronymic",
      "base": "Erik",
      "eras": ["viking", "medieval", "classic"],
      "meaning": "ever-powerful ruler"
    }
  ]
}
```

**Era values:** `"viking"` | `"medieval"` | `"classic"` | `"modern"`
**Gender values:** `"m"` | `"f"`
**Patronymic display logic (computed in JS, never stored):**
- Male → `base + "sen"` (e.g. base "Erik" → "Eriksen")
- Female → `base + "datter"` (e.g. base "Erik" → "Eriksdatter")

## Task 1: Write the failing data test

**Files:**
- Create: `tests/danish-names-data.test.js`

- [ ] **Step 1: Write the test**

```js
const assert = require('assert');
const fs = require('fs');
const path = require('path');

const file = path.join(__dirname, '..', 'wordineer-deploy', 'data', 'danish-names.json');
assert.ok(fs.existsSync(file), 'danish-names.json should exist');

const data = JSON.parse(fs.readFileSync(file, 'utf8'));

assert.ok(Array.isArray(data.given), 'given should be an array');
assert.ok(Array.isArray(data.surnames), 'surnames should be an array');
assert.ok(data.given.length >= 150, `expected at least 150 given names, got ${data.given.length}`);
assert.ok(data.surnames.length >= 100, `expected at least 100 surnames, got ${data.surnames.length}`);

const VALID_ERAS = new Set(['viking', 'medieval', 'classic', 'modern']);

for (const item of data.given) {
  assert.strictEqual(typeof item.name, 'string', 'given name must be a string');
  assert.ok(/^\p{Lu}[\p{L}' -]+$/u.test(item.name), `given name should be title case: ${item.name}`);
  assert.ok(['m', 'f'].includes(item.gender), `invalid gender: ${item.gender} on ${item.name}`);
  assert.ok(Array.isArray(item.eras) && item.eras.length > 0, `${item.name} must have eras array`);
  for (const e of item.eras) assert.ok(VALID_ERAS.has(e), `invalid era "${e}" on ${item.name}`);
  assert.strictEqual(typeof item.meaning, 'string', `${item.name} meaning must be a string`);
  assert.ok(item.meaning.length > 0, `${item.name} meaning must not be empty`);
}

const genderSet = new Set(data.given.map(n => n.gender));
assert.ok(genderSet.has('m'), 'dataset must have male given names');
assert.ok(genderSet.has('f'), 'dataset must have female given names');

const eraSet = new Set(data.given.flatMap(n => n.eras));
for (const e of ['viking', 'medieval', 'classic', 'modern']) {
  assert.ok(eraSet.has(e), `given names must cover era: ${e}`);
}

let fixedCount = 0;
let patronymicCount = 0;

for (const item of data.surnames) {
  assert.ok(['fixed', 'patronymic'].includes(item.type), `invalid surname type: ${item.type}`);
  assert.ok(Array.isArray(item.eras) && item.eras.length > 0, 'surname must have eras');
  for (const e of item.eras) assert.ok(VALID_ERAS.has(e), `invalid era "${e}" in surname`);
  assert.strictEqual(typeof item.meaning, 'string', 'surname meaning must be a string');
  assert.ok(item.meaning.length > 0, 'surname meaning must not be empty');

  if (item.type === 'fixed') {
    assert.strictEqual(typeof item.name, 'string', 'fixed surname must have name');
    assert.ok(/^\p{Lu}[\p{L}' -]+$/u.test(item.name), `fixed surname should be title case: ${item.name}`);
    fixedCount++;
  } else {
    assert.strictEqual(typeof item.base, 'string', 'patronymic surname must have base');
    assert.ok(item.base.length > 0, 'patronymic base must not be empty');
    patronymicCount++;
  }
}

assert.ok(fixedCount >= 60, `expected at least 60 fixed surnames, got ${fixedCount}`);
assert.ok(patronymicCount >= 40, `expected at least 40 patronymic bases, got ${patronymicCount}`);

console.log(`danish-names.json ok: ${data.given.length} given (${[...genderSet]}) | ${fixedCount} fixed + ${patronymicCount} patronymic surnames`);
```

  <h2>Danish Naming History</h2>
  <p>Danish names trace directly back to Old Norse, the language spoken across Scandinavia during the Viking Age (roughly 800–1100 AD). Names from this period are often compound forms built from Norse roots: <em>bjørn</em> (bear), <em>sig</em> (victory), <em>ulf</em> (wolf), <em>hild</em> (battle). The goddess Freya, the thunder god Thor, and the concept of victory (<em>sig-</em>) all appear directly in Viking given names.</p>
  <p>During the medieval period, Christianity arrived and brought a wave of biblical and Latin names — Jens (John), Peder (Peter), Mads (Matthew), Karen (Catherine) — which gradually blended with the Norse tradition. Surnames during this era were still patronymic: a man took his father's first name and added <em>-sen</em> (son of), while a woman added <em>-datter</em> (daughter of). So Erik's son was Eriksen and Erik's daughter was Eriksdatter.</p>
  <p>The Surname Act of 1856 required Danish families to choose a fixed hereditary surname, effectively freezing what had been fluid patronymics into permanent family names. This is why the most common Danish surnames — Jensen, Nielsen, Hansen, Pedersen — all end in -sen and were once patronymics. A 2006 law later re-allowed the traditional patronymic system for parents who want to use it.</p>

const DNG_ERA_LABELS = {
  viking: 'Viking Age',
  medieval: 'Medieval',
  classic: '19th Century',
  modern: 'Modern'
};
const dngsen = 'sen';
const dngdatter = 'datter';

let dngData = { given: [], surnames: [] };
let dngReady = false;
let dngCurrentItems = [];

function dngEsc(s) {
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}

function dngShuffle(arr) {
  const a = arr.slice();
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

function dngCount() {
  const v = parseInt(document.getElementById('dng-count')?.value, 10);
  if (isNaN(v)) return 10;
  return Math.max(1, Math.min(50, v));
}

function dngHasEra(eras, selected) {
  return selected === 'any' || (Array.isArray(eras) && eras.includes(selected));
}

function dngDefsVisible() {
  return document.getElementById('dng-defs')?.checked !== false;
}

function dngNormalizeGiven(entry) {
  if (!entry || typeof entry !== 'object') return null;
  const name = String(entry.name || '').trim();
  if (!name) return null;
  const gender = String(entry.gender || 'm').trim();
  const eras = Array.isArray(entry.eras) ? entry.eras : [];
  const meaning = String(entry.meaning || '').trim();
  return [name, gender, eras, meaning];
}

function dngNormalizeSurname(entry) {
  if (!entry || typeof entry !== 'object') return null;
  const type = String(entry.type || 'fixed').trim();
  const eras = Array.isArray(entry.eras) ? entry.eras : [];
  const meaning = String(entry.meaning || '').trim();
  if (type === 'patronymic') {
    const base = String(entry.base || '').trim();
    if (!base) return null;
    return { type: 'patronymic', base, eras, meaning };
  }
  const name = String(entry.name || '').trim();
  if (!name) return null;
  return { type: 'fixed', name, eras, meaning };
}

function dngResolveSurname(s, gender) {
  if (s.type === 'patronymic') {
    return gender === 'f' ? s.base + dngdatter : s.base + dngsen;
  }
  return s.name;
}

function dngNoResultsItem() {
  return {
    display: 'No matching Danish names found',
    meaning: 'Try a broader era, gender, or starting letter.',
    era: '',
    noResult: true
  };

function dngCurrentFilterValues() {

    gender: document.getElementById('dng-gender')?.value || 'any',
    era: document.getElementById('dng-era')?.value || 'any',
    letter: document.getElementById('dng-letter')?.value || 'any'
  };
}

function dngGenerateFn(data) {
  if (!dngReady || !data.given.length || !data.surnames.length) {
    return [{ display: 'Loading Danish names...', meaning: 'Please wait while the name data loads.', era: '' }];
  }

  const filters = dngCurrentFilterValues();
  const limit = dngCount();
  const letter = filters.letter === 'any' ? '' : filters.letter.toUpperCase();

  const givenPool = data.given.filter(function(n) {
    if (filters.gender !== 'any' && n[1] !== filters.gender) return false;
    if (!dngHasEra(n[2], filters.era)) return false;
    if (letter && !n[0].toUpperCase().startsWith(letter)) return false;
    return true;
  });

  const surnamePool = data.surnames.filter(function(s) {
    if (!dngHasEra(s.eras, filters.era)) return false;
    return true;
  });

  if (filters.type === 'given' && !givenPool.length) return [dngNoResultsItem()];
  if (filters.type === 'surname' && !surnamePool.length) return [dngNoResultsItem()];
  if (filters.type === 'full' && (!givenPool.length || !surnamePool.length)) return [dngNoResultsItem()];

  const sg = dngShuffle(givenPool);
  const ss = dngShuffle(surnamePool);
  const results = [];
  const seen = new Set();
  let gi = 0;
  let si = 0;
  let guard = 0;

  while (results.length < limit && guard < limit * 40) {
    guard++;

    if (filters.type === 'given') {
      const given = sg[gi++ % sg.length];
      if (!given) break;
      const display = given[0];
      if (seen.has(display)) continue;
      seen.add(display);
      const bestEra = given[2][0] || '';
      results.push({ display, meaning: given[3], era: bestEra, kind: 'given' });
      continue;
    }

    if (filters.type === 'surname') {
      const sur = ss[si++ % ss.length];
      if (!sur) break;
      const resolvedGender = filters.gender === 'any' ? 'm' : filters.gender;
      const display = dngResolveSurname(sur, resolvedGender);
      if (letter && !display.toUpperCase().startsWith(letter)) continue;
      if (seen.has(display)) continue;
      seen.add(display);
      const bestEra = sur.eras[0] || '';
      results.push({ display, meaning: sur.meaning, era: bestEra, kind: 'surname' });
      continue;
    }

    const given = sg[gi++ % sg.length];
    const sur = ss[si++ % ss.length];
    if (!given || !sur) break;
    if (letter && !given[0].toUpperCase().startsWith(letter)) continue;
    const genderForSurname = given[1];
    const surname = dngResolveSurname(sur, genderForSurname);
    const display = given[0] + ' ' + surname;
    if (seen.has(display)) continue;
    seen.add(display);
    const bestEra = given[2][0] || sur.eras[0] || '';
    const meaning = given[3] && sur.meaning ? given[3] + '; ' + sur.meaning : (given[3] || sur.meaning || '');
    results.push({ display, meaning, era: bestEra, kind: 'full' });
  }

  dngCurrentItems = results.slice();
  return results.length ? results : [dngNoResultsItem()];
}

function dngRenderItem(item) {
  if (item.noResult) {

  let savedArr = [];

  const isSaved = savedArr.includes(item.display);
  const safeAttr = item.display.replace(/"/g, '&quot;');
  const safeClick = item.display.replace(/\\/g, '\\\\').replace(/'/g, "\\'");

  return '<li class="word-item">'

    +   '<div class="word-text">' + dngEsc(item.display) + '</div>'
    +   '<div class="dng-badges">' + eraLabel + typeLabel + '</div>'
